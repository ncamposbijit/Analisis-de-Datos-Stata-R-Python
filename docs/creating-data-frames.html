5.1 Creating Data Frames

# Create a data frame
df <- data.frame(
  ID = 1:5,
  Name = c("Alice", "Bob", "Charlie", "Diana", "Eve"),
  Age = c(25, 30, 35, 28, 32),
  Employed = c(TRUE, TRUE, FALSE, TRUE, TRUE)
)

df
